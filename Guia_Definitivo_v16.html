local function getCapOz() local ok, v = pcall(Player.getCapacity); return (ok and v) and v / 100 or 0 end
local function countItem(id) if not id then return 0 end local ok, v = pcall(Game.getItemCount, id); return ok and v or 0 end
local function getPos() local ok, p = pcall(Map.getCameraPosition); return ok and p or { x = 0, y = 0, z = 0 } end
local function nearPos(a, b, tol) if not a or not b or a.z ~= b.z then return false end return math.max(math.abs(a.x - b.x), math.abs(a.y - b.y)) <= (tol or 1) end

local function getDistance(pos1, pos2)
    if not pos1 or not pos2 then return 999 end
    return math.max(math.abs(pos1.x - pos2.x), math.abs(pos1.y - pos2.y))
end

-- =============================================

local function sendFollow(targetName)
    pcall(Player.stopAutoWalk)
    pcall(Game.talk, "!follow " .. targetName, Enums.TalkTypes.TALKTYPE_SAY)
end

local function sendUnfollow()
    pcall(Game.talk, "!follow", Enums.TalkTypes.TALKTYPE_SAY)
end

-- =============================================
-- GPS EMERGÊNCIA → TILE ADJACENTE
-- =============================================
local function goToAdjacentTile(targetX, targetY, targetZ)
    local myPos = getPos()
    if myPos.z ~= targetZ then
        pcall(Map.goTo, targetX, targetY, targetZ)
        return
    end

    local dx = targetX - myPos.x
    local dy = targetY - myPos.y
    local adjX, adjY = targetX, targetY

    if math.abs(dx) >= math.abs(dy) then
        if dx > 0 then adjX = targetX - 1
        elseif dx < 0 then adjX = targetX + 1 end
    else
        if dy > 0 then adjY = targetY - 1
        elseif dy < 0 then adjY = targetY + 1 end
    end

    if adjX == targetX and adjY == targetY then
        adjY = targetY + 1
    end

    pcall(Map.goTo, adjX, adjY, targetZ)
end

-- =============================================
-- DETECÇÃO DE MORTE
-- =============================================
local function amIDead()
    local ok, hp = pcall(Player.getHealth)
    if ok and hp ~= nil and hp <= 0 then return true end
    return false
end

local function needsRefill()
    if getCapOz() < CAP_MIN then return true, "Cap" end
    for _, item in ipairs(myVoc) do
        if countItem(item.id) < item.min then return true, item.name end
    end
    return false, "OK"
end

local function renderHud(isLost, needR, reason, followStatus, dist)
    local myPos = getPos()

    )
    for _, item in ipairs(myVoc) do text = text .. string.format("%s: %d\n", item.name, countItem(item.id)) end
    if not hudHandle then hudHandle = HUD.new(10, 10, text, true); hudHandle:setDraggable(true) else hudHandle:setText(text) end
    hudHandle:setColor(220, 220, 220)

print("[FOLLOWER v19] Fase dita tudo. Sem geofence. !follow + GPS emergencia.")
local isFollowing    = false
local gpsActive      = false
local myPulse        = 0
local cbState        = nil
local lastPhase      = "STANDBY"
local refollowTimer  = 0
local isDead         = false
local followStatus   = "NENHUM"

-- =============================================
-- LOOP PRINCIPAL
-- =============================================
while true do
    myPulse = myPulse + 1
    local st = Comms.FollowerReceive()
    if st and st.phase then _G.FollowerPhase = st.phase end

    local myPos  = getPos()
    local isLost = false
    isDead = amIDead()

    -- Posição do líder via UDP
    local leaderDist   = 999
    local leaderFloor  = myPos.z
    local hasLeaderPos = false
    if st and st.leaderX and st.leaderY and st.leaderZ then
        leaderDist   = getDistance(myPos, { x = st.leaderX, y = st.leaderY, z = st.leaderZ })
        leaderFloor  = st.leaderZ
        hasLeaderPos = true
    end

    local diffFloor   = (leaderFloor ~= myPos.z)
    local extremeDist = (leaderDist > GPS_DIST_EXTREME and not diffFloor)

    -- =========================================
    -- BLOCO A: FASES DE FOLLOW (HUNTING / WAIT_SAFE)
    -- =========================================
    if FOLLOW_PHASES[_G.FollowerPhase] then

        if not FOLLOW_PHASES[lastPhase] then
            sendFollow(TARGET_TO_FOLLOW)
            isFollowing   = true
            gpsActive     = false
            refollowTimer = 0
            followStatus  = "!follow " .. TARGET_TO_FOLLOW
        end

        if diffFloor or extremeDist then
            isLost = true

            if not gpsActive then
                sendUnfollow()
                isFollowing = false
                gpsActive   = true
            end

            followStatus = "GPS EMERGENCIA"

            if hasLeaderPos then
                goToAdjacentTile(st.leaderX, st.leaderY, st.leaderZ)
            end
        else
            if gpsActive then
                pcall(Player.stopAutoWalk)
                sendFollow(TARGET_TO_FOLLOW)
                gpsActive     = false
                isFollowing   = true
                refollowTimer = 0
                followStatus  = "!follow " .. TARGET_TO_FOLLOW .. " (retomado)"
            elseif not isFollowing then
                sendFollow(TARGET_TO_FOLLOW)
                isFollowing   = true
                refollowTimer = 0
                followStatus  = "!follow " .. TARGET_TO_FOLLOW
            else
                refollowTimer = refollowTimer + 1
                if refollowTimer >= REFOLLOW_INTERVAL then
                    sendFollow(TARGET_TO_FOLLOW)
                    refollowTimer = 0
                    followStatus = "!follow " .. TARGET_TO_FOLLOW .. " (refresh)"
                end
            end
        end

        if cbState ~= false then pcall(Engine.enableCaveBot, false); cbState = false end

    -- =========================================
    -- BLOCO B: FASES DE ROTA (REFILL)
    -- =========================================
    elseif ROUTE_PHASES[_G.FollowerPhase] then
        if isFollowing then
            sendUnfollow()
            isFollowing = false
        end
        if gpsActive then
            pcall(Player.stopAutoWalk)
            gpsActive = false
        end
        refollowTimer = 0
        followStatus  = "ROTA CAVEBOT"
        if cbState ~= true then pcall(Engine.enableCaveBot, true); cbState = true end

    -- =========================================
    -- BLOCO C: STANDBY
    -- =========================================
    else
        if isFollowing then
            sendUnfollow()
            isFollowing = false
        end
        if gpsActive then
            pcall(Player.stopAutoWalk)
            gpsActive = false
        end
        refollowTimer = 0
        followStatus  = "PARADO (STANDBY)"
        cbState       = nil
    end

    lastPhase = _G.FollowerPhase

    -- =========================================
    -- COMUNICAÇÃO
    -- =========================================
    saveCavebotPhase(_G.FollowerPhase)
    local needR, reason = needsRefill()
    saveFollowerNeed(needR)
    
    Comms.FollowerSendToLeader({ 
        from = MY_NAME, needsRefill = needR, reason = reason,
        atSafe = nearPos(getPos(), SAFE_POS, POS_TOLERANCE),
        cap = getCapOz(), fPhase = _G.FollowerPhase,
        isFollowing = isFollowing, isLost = isLost, pulse = myPulse,
        isDead = isDead
    })
    
    renderHud(isLost, needR, reason, followStatus, leaderDist)
    wait(200)
end

print("[LEADER v17] Radio HUD + Pos Tracker + Detecção de Morte + Emergência.")
local radarBraked       = false
local emergencyActive   = false  
local emergencyTimer    = 0      

while true do
    readCavebotPhase()

    for _, f in ipairs(FOLLOWERS) do
        local pkt = Comms.LeaderReceiveFrom(f.name)
        if pkt and pkt.from and flags[pkt.from] and pkt.pulse ~= flags[pkt.from].last_pulse then 
            local fl = flags[pkt.from]
            fl.needs_refill = pkt.needsRefill
            fl.at_safe      = pkt.atSafe
            fl.is_following = pkt.isFollowing
            fl.is_lost      = pkt.isLost
            fl.last_seen    = os.time()
            fl.last_pulse   = pkt.pulse

            if pkt.isDead then
                if not fl.is_dead then
                    fl.is_dead = true
                    fl.dead_since = os.time()
                end
            else
                if fl.is_dead then
                    fl.is_dead = false
                    fl.dead_since = 0
                end
            end
        end
    end

    local lNeed = leaderNeedsRefill()
    local fNeed, allAtSafe, allFollowing, anyLost, anyDead = false, nearPos(getPos(), SAFE_POS, POS_TOLERANCE), true, false, false

    for _, f in ipairs(FOLLOWERS) do
        local fl = flags[f.name]
        local isOnline = (os.time() - fl.last_seen) < OFFLINE_TIMEOUT

        if fl.is_dead then
            anyDead = true
        end

        if isOnline and not fl.is_dead then 
            if fl.needs_refill then fNeed = true end
            if not fl.at_safe then allAtSafe = false end
            if not fl.is_following then allFollowing = false end
            if fl.is_lost then anyLost = true end
        elseif not isOnline then
            allAtSafe = false; allFollowing = false; anyLost = true
            if _G.LeaderPhase == "HUNTING" and fl.last_seen > 0 then
                anyDead = true
            end
        end
    end

    if anyDead and _G.LeaderPhase == "HUNTING" and not emergencyActive then
        emergencyActive = true
        emergencyTimer  = os.time()
        forcePhase("WAIT_SAFE")
    end

    if emergencyActive and not anyDead and allAtSafe then
        emergencyActive = false
        emergencyTimer  = 0
    end

    if _G.LeaderPhase == "HUNTING" then
        if anyLost or anyDead then
            if not radarBraked then 
                pcall(Engine.enableCaveBot, false)
                radarBraked = true
            end
        else
            if radarBraked then pcall(Engine.enableCaveBot, true); radarBraked = false end
        end
    else
        if radarBraked then radarBraked = false end
    end

    saveCavebotState(lNeed, fNeed, allAtSafe, allFollowing)
    Comms.LeaderBroadcast(FOLLOWERS, { 
        phase   = _G.LeaderPhase, 
        leaderX = getPos().x, 
        leaderY = getPos().y, 
        leaderZ = getPos().z 
    })
    renderHud(lNeed, fNeed, allAtSafe, allFollowing, emergencyActive)
    wait(200)
end
